用法说明

以下组件使用 JSON 管道将对象转换为 JSON 格式，并以两种格式显示字符串以进行比较。

@Component({  selector: 'json-pipe',  template: `<div>    <p>Without JSON pipe:</p>    <pre>{{ object }}</pre>    <p>With JSON pipe:</p>    <pre>{{ object | json }}</pre>  </div>`,  standalone: false,})export class JsonPipeComponent {  object: Object = {foo: 'bar', baz: 'qux', nested: {xyz: 3, numbers: [1, 2, 3, 4, 5]}};}
